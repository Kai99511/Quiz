this.$questionButton.on("click", function () {
  // ユーザーの選択を取得
  const selectedChoice = $(this).hasClass('button01') ? 'answer01' :
                         $(this).hasClass('button02') ? 'answer02' :
                         $(this).hasClass('button03') ? 'answer03' :
                         $(this).hasClass('button04') ? 'answer04' : null;

  // 正解かどうかの判定
  if (selectedChoice !== null && selectedChoice === 'answer01') {
    // 正解の場合
    $(this).parents('.quiz-answer').addClass('is-correct');
    score += $pointPerCorrect;
  } else {
    // 不正解の場合
    $(this).parents('.quiz-answer').addClass('is-incorrect');
  }

  // クリックされたボタンにもクラスを追加
  $(this).addClass('is-checked');

  return false;
});
